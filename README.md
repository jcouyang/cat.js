## Install
### Browser
```html
<script src="//cdn.rawgit.com/jcouyang/cat.js/master/dist/cat.min.js"></script>
```

### Node/Browserify
```js
let {maybe, just} = require('cat.js');
maybe(false, Boolean, just(2))
```
